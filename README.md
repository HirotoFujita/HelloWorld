```
git clone https://github.com/buntyke/HelloWorld.git
ls
cd HelloWorld/
ls
nano Nishanth.md
ls
git status
git add Nishanth.md 
git status
git commit -m 'commit by Nishanth'
git status
git log
git remote -v
git push -u origin master
```
